    end_date date,

    aws_lambda double precision,
    aws_secrets_manager double precision,
    amazon_api_gateway double precision,
    amazon_cloudfront double precision,
    amazon_ec2_container_registry_ecr double precision,
    ec2_other double precision,
    amazon_elastic_compute_cloud_compute double precision,
    amazon_elastic_container_service double precision,
    amazon_elastic_container_service_for_kubernetes double precision,
    amazon_elastic_load_balancer double precision,
    amazon_relational_database_service double precision,
    amazon_simple_email_service double precision,
    amazon_simple_notification_service double precision,
    amazon_simple_storage_service double precision,
    amazon_cloudwatch double precision,
    aws_cost_explorer double precision,
    tax double precision
)
